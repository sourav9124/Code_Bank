 // This solution is being solved using below steps : 
 step 1 => Insert the elements into the set.
 step 2 => Then put back the the set elements into the orioginal array initiating with 0.
 step 3 => return the size of the modified original array.

 int removeDuplicates(vector<int>& nums) {
      
      int n=nums.size();
      set<int> st;

      for(int i=0;i<n;i++)
      {
        st.insert(nums[i]);
      }
      int i=0;
      for(auto k:st)
      {
        nums[i]=k;
        i++;
      }
      return i;

    }
